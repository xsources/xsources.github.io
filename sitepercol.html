Interactive site percolation in square lattice

We present here a percolation system in a (100x100) square lattice. Our interactive python application computes and updates at each click on the "Next" button the following elements:

  • the probability (p)
  • the concentration (c) of “occupied sites”
  • the size of the lagest component (LC), colored in red
    At the percolation threshold pc=0.5927, the largest component spans the percolation system by joining two opposite sides of the square grid.

def click(b):
    global cc;    cc +=0.02;    B = np.where(R < cc, 1, 0);    B1=logical_not(B)*1;    lw, num = measurements.label(B);    w=np.max(lw);    area = measurements.sum(B, lw, index=arange(lw.max() + 1));    sB=B.sum();    w1=np.max(area);    ix=[i for i, x in enumerate(area) if x == w1];    a = np.array(ix);        C = np.where(lw ==a, 1, 0);        C1 = logical_not(C)*1;    D=B1+C1;        ama = logical_not(D)*1;        sama=ama.sum();        p=sB/(N*N);     im = ax1.imshow(D, cmap=cmap, interpolation='nearest'); 
    with out:
        ax2.plot(p,sama/sB,'ro-');        display(ax1.figure);        print("p= ",p, "c= ",sB,"LC= ",sama," Author: S.Boutiche");        clear_output(wait=True);
button.on_click(click);
click(None);
